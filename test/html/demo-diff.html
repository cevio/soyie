// getTextNode 被认为获取文本节点即可
// scope 为数据源
var textnode = getTextNode('name');
$("[es-binding='name']").on('input', function(){
    var val = $(this).val();
    textnode.nodeValue = val;
    scope['name'] = val;
});
//observe 被认为数据监听方法
observe(scope, function(name, newvalue){
    if ( name === 'name' ){
        $("[es-binding='name']").val(newvalue);
    }
});
scope.name = 'XYBK';
